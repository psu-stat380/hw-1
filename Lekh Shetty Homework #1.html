row_wise_scan <- function(x){
    n <- nrow(x)
    m <- ncol(x)

    # Insert your code here
    count <- 0
    for(i in 1:n){
        for(j in 1:m){
            if(x[i,j] >= 0){
                count <- count + 1 
            }
        }
    }

    return(count)
}

col_wise_scan <- function(x){
    count <- 0
    
    # Insert your code here
    n <- nrow(x)
    m <- ncol(x)
    for(j in 1:m){
      for(i in 1:n){
        if(x[i,j] >= 0){
            count <- count + 1 
        }
    }
}

    return(count)
}
